Quiz = ({
    "How many spaces can a pawn move on its first turn?": "2",
    "How many spaces can a pawn move on its second turn?": "1",
    "How many pawns exist on one side inn a chess game?": "8"
})
correct = 0
for key, value in Quiz.items():
    questions = input(key)
    if questions == value:
        print("Correct")
        correct+=1
    else:
        print("Incorrect")
print("Score: ",correct)
